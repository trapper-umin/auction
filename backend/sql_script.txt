CREATE TABLE person(
    person_id int PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY ,
    name varchar NOT NULL UNIQUE CHECK ( length(name)>3 AND length(name)<255 ),
    created_at TIMESTAMP NOT NULL,
    updated_at TIMESTAMP NOT NULL
);

CREATE TABLE creator(
    creator_id int PRIMARY KEY GENERATED BY DEFAULT AS IDENTITY ,
    name varchar NOT NULL CHECK ( length(name)>=3 AND length(name)<=255),
    description varchar NOT NULL CHECK ( length(description)>=3 AND length(description)<=255),
    created_at timestamp NOT NULL ,
    updated_at timestamp NOT NULL
)
